# helper functions for text preprocessing
def tokenize(text):
    return nltk.word_tokenize(text)

def remove_numbers(text):
    return re.sub(r'\d+', '', text)

def remove_urls(text):
    return re.sub(_URL_PATTERN, '', text)

def remove_stopwords(text, lang='english'):
    """
    :param text: text input
    :return: text with stopwords removed
    :rtype: str 
    """
    words = nltk.word_tokenize(text)
    lang_stopwords= [i for i in stopwords.words(lang) if i not in ['not', 'no']]
    stopwords_removed= [w for w in words if w not in lang_stopwords]
    return " ".join(stopwords_removed)

def expand_contractions(text) :
    pattern = re.compile("({})".format("|".join(_CONTRACTION_MAP.keys())),flags = re.DOTALL| re.IGNORECASE)
    
    def replace_text(t):
        txt = t.group(0)
        if txt.lower() in _CONTRACTION_MAP.keys():
            return _CONTRACTION_MAP[txt.lower()]
        
    expand_text = pattern.sub(replace_text,text)
    return expand_text 

# Function to apply lemmatization to a list of words
def words_lemmatizer(text, encoding="utf8"):
    words = nltk.word_tokenize(text)
    lemma_words = []
    wl = WordNetLemmatizer()
    for word in words:
        pos = find_pos(word)
        lemma_words.append(wl.lemmatize(word, pos))
    return " ".join(lemma_words)

def stemmer(text):
    porter_stemmer=nltk.PorterStemmer()
    words = [porter_stemmer.stem(word) for word in nltk.word_tokenize(text)]
    return " ".join(words)

# Function to find part of speech tag for a word
def find_pos(word):
    # Part of Speech constants
    # ADJ, ADJ_SAT, ADV, NOUN, VERB = 'a', 's', 'r', 'n', 'v'
    pos = nltk.pos_tag(nltk.word_tokenize(word))[0][1]
    
    # Adjective tags - 'JJ', 'JJR', 'JJS'
    if pos.lower()[0] == 'j':
        return 'a'
    # Adverb tags - 'RB', 'RBR', 'RBS'
    elif pos.lower()[0] == 'r':
        return 'r'
    # Verb tags - 'VB', 'VBD', 'VBG', 'VBN', 'VBP', 'VBZ'
    elif pos.lower()[0] == 'v':
        return 'v'
    # Noun tags - 'NN', 'NNS', 'NNP', 'NNPS'
    else:
        return 'n'
def normalize(text):
    words = nltk.word_tokenize(text)
    return ' '.join(normalise(words, verbose=False))

def trim(text):
    words = nltk.word_tokenize(text)
    clean_words = []
    for i in words:
        if re.search('^[^\w\s]+$', i) or re.search('^_+$', i):
            continue
        elif re.search('[\w\s]',i) and re.search('[^\w\s]|_', i):
            clean_words.append(re.sub('[^\w\s]|_','',i))
        else:
            clean_words.append(i)
    return " ".join([i for i in clean_words if len(i)>=2 and len(i)<=15])

def compute_weight(text):
    _keywords_dict = dict(zip(_keywords, _keywords_wgh))
    words = nltk.word_tokenize(text)
    keywords = [i for i in words if i in _keywords_dict.keys()]
    weight = 0
    for i in keywords:
        weight+=_keywords_dict.get(i)
    return weight

def do_preprocessing(df):
    """
    create handcrafted features
    """
    df['Comment_cleaned'] = df['Comment'].str.lower().\
                                apply(expand_contractions).\
                                apply(remove_urls).\
                                str.replace('[^\w\s]|_', ' ').\
                                apply(remove_numbers).\
                                apply(remove_stopwords)
    df['Comment_cleaned_lemma'] = df['Comment_cleaned'].apply(words_lemmatizer)
    df['Comment_cleaned_stem'] = df['Comment_cleaned'].apply(stemmer)
    # hand-crafted features
    df['words'] = df['Comment'].apply(tokenize)
    df['text__len'] = df['words'].apply(len)
    df['sent__num'] = df['Comment'].apply(lambda x: len(nltk.sent_tokenize(x)))
    df['avgWrd__len'] = df['words'].apply(lambda words: round(np.average([len(wrd) for wrd in words]),2))
    df['url__cnt'] = df['Comment'].apply(lambda x: len(re.findall(_URL_PATTERN, x)))
    df['digit__cnt'] = df['words'].apply(lambda x: sum([re.search('\d', i) is not None for i in x]))
    df['bracket__cnt'] = df['words'].apply(lambda x: sum([re.search('\(|\)|\[|\]', i) is not None for i in x]))
    df['equal__cnt'] = df['Comment'].apply(lambda x: len(re.findall('=|<-', x)))
    df['verb__cnt'] = df['Comment'].apply(lambda x: sum([tag[0]=='V' for token, tag in TextBlob(x).tags]))
    df['noun__cnt'] = df['Comment'].apply(lambda x: sum([tag[0]=='N' for token, tag in TextBlob(x).tags]))
    df['adv__cnt'] = df['Comment'].apply(lambda x: sum([tag[0]=='R' for token, tag in TextBlob(x).tags]))
    df['adj__cnt'] = df['Comment'].apply(lambda x: sum([tag[0]=='J' for token, tag in TextBlob(x).tags]))
    df['at__user'] = df['Comment'].apply(lambda x: len(re.findall('@', x)))
    df['comment__cnt'] = df['Comment'].apply(lambda x: len(re.findall('#', x)))
    df['libKeywords__cnt'] = df['Comment'].apply(lambda x: len(re.findall('library|package', x)))
    df['codeMethod__cnt']  = df['Comment'].apply(remove_urls).apply(lambda x: len(re.findall('[0-9a-z]{2,}\.[0-9a-z]{2,}?|[0-9a-z]{2,}\([0-9a-z]{2,}?', x)))
    df['keywrds__weight'] = df['Comment_cleaned_lemma'].apply(lambda x: compute_weight(x))    
    df['nonStop__cnt'] = df['words'].apply(lambda x: len([i for i in x if i not in stopwords.words('english')]))
    df['continuousChar__cnt'] = df['Comment'].apply(lambda x: len([match.group() for match in re.compile(r'([a-z])\1{2,}').finditer(x)]))
    df['continuousDigit__cnt'] = df['Comment'].apply(lambda x: len([match.group() for match in re.compile(r'([0-9])\1{2,}').finditer(x)]))
    df['continuousPunct__cnt'] = df['Comment'].apply(lambda x: len([match.group() for match in re.compile(r'([\W|_])\1{2,}').finditer(x)]))
    for i in standard_chars:
        df[i+'__cnt'] = df['Comment'].apply(lambda x: len(re.findall(re.escape(i), x)))
    for i in _punct:
        pattern = "(\\" + i + ")\\1{2,}"
        df[i + '__continuous__cnt'] = df['Comment'].apply(lambda x: len([match.group() for match in re.compile(pattern).finditer(x)]))
    return df

def model_fit_prediction(model, train_X, train_Y, test_X, test_Y, kfold=5, is_trained=False):
    '''
    model training and predicting
    model - model object
    train_X, test_X, train_Y, test_Y - train/test data after train/test split
    '''

    # fit a model
    if not is_trained:
        model.fit(train_X, train_Y)
    pred_Y = model.predict(test_X)
    print('Accuracy on training set: {:.4f}'.format(model.score(train_X, train_Y)))
    print('CV Accuracy on training set: {:.4f}'.format(cross_val_score(model, train_X, train_Y, cv=kfold, scoring='accuracy').mean()))

    # accuracy score
    print('Accuracy on test set: {:.4f}'.format(model.score(test_X, test_Y)))
     
    # plot confusion matrix
    plot_confusion_matrix(test_Y, pred_Y)
    
    # AUC
    print('AUC: %.10f' %roc_auc_score(test_Y,pred_Y))
    plot_roc_curve(model, test_X, test_Y)
    
    # classification report
    print(classification_report(test_Y, pred_Y, digits=4))
    
    result =  {
        'train_X': train_X, 
        'test_X' : test_X,
        'model': model,
        'prediction': pred_Y
    }
    return result

def modelfit(model,train_X, train_Y ,cv_folds, useTrainCV=True, early_stopping_rounds=50):
    '''
    find appropriate number of boosting rounds (n_estimators)
    '''
    if useTrainCV:
        xgb_param = model.get_xgb_params()
        xgtrain = xgb.DMatrix(train_X, label=train_Y)
        cvresult = xgb.cv(xgb_param, xgtrain, num_boost_round=model.get_params()['n_estimators'], folds=cv_folds,
            metrics='error', early_stopping_rounds=early_stopping_rounds, verbose_eval=50)
        model.set_params(n_estimators=cvresult.shape[0])
    
    #Fit the model on the data
    model.fit(train_X, train_Y)
    
    #Predict training set:
    train_predictions = model.predict(train_X)   
    print("\nModel Parameters")
    print(model.get_xgb_params())
    #Print model report:
    print("\nModel Report")
    print("Accuracy : %.4g" % accuracy_score(train_Y, train_predictions))    
    return model, cvresult

def create_model_nn():
    word_input = Input(shape=(maxlen,), dtype='int32',name='words_input')
    feature_input = Input(shape=(feature_no,), dtype='float32',name='features_input')
    words = Embedding(max_words, 100, input_length=maxlen)(word_input)
    # words = Embedding(len(word_index) + 1,
    #                  EMBEDDING_DIM,
    #                  weights=[embedding_matrix],
    #                  input_length=maxlen,
    #                  trainable=False)(word_input)
    output = Conv1D(64, 3, activation='relu', padding='same')(words)
    output = MaxPooling1D(2)(output)
    output= Dropout(0.5)(output)
    output = Conv1D(64, 3, activation='relu', padding='same')(output)
    output = Bidirectional(LSTM(50, dropout=0.2, recurrent_dropout=0.5))(output)
    output = concatenate([output, feature_input])
    output = Dense(30, activation='relu')(output)
    output= Dropout(0.5)(output)
    output = Dense(1, activation='sigmoid')(output)
    model_nn = Model(inputs=[word_input, feature_input], outputs=[output])
    opt = optimizers.RMSprop(lr=0.0005)
    model_nn.compile(
        loss='binary_crossentropy', 
        optimizer=opt, 
        metrics = ['accuracy'])
    return model_nn

def stacking(model,train_X,train_Y,test_X, test_Y, n_fold):
    """
    1st level stacking for xgboost/lightgbm by kfold training and prediction
    """
    folds=StratifiedKFold(n_splits=n_fold,random_state=2020, shuffle=True)
    test_pred=np.empty((0,1),float)
    train_pred=np.empty((0,1),float)
    train_y_shuffled = np.empty((0,1),float)
    for train_indices,val_indices in folds.split(train_X,train_Y):
        print(val_indices[[0,-1]])
        x_train,x_val=train_X[train_indices],train_X[val_indices]
        y_train,y_val=train_Y[train_indices],train_Y[val_indices]
        train_y_shuffled = np.append(train_y_shuffled, y_val)
        evaluation = [(x_train, y_train), (x_val, y_val)]
        model.fit(x_train, y_train, eval_set=evaluation, eval_metric="logloss", early_stopping_rounds=200,verbose=False)
        train_pred=np.append(train_pred,model.predict_proba(x_val)[:, 1])
    evaluation = [(train_X, train_Y), (test_X, test_Y)]
    model.fit(train_X, train_Y, eval_set=evaluation, eval_metric="logloss", early_stopping_rounds=200,verbose=False)
    test_pred=np.append(test_pred,model.predict_proba(test_X)[:, 1])
    return train_pred, train_y_shuffled, test_pred

def stacking_nn(model,train_X_word, train_X_feature, train_Y, test_X_word, test_X_feature, n_fold):
    """
    1st level stacking for neural network by kfold training and prediction
    """
    epochs = 100
    batch_size = 32
    folds=StratifiedKFold(n_splits=n_fold,random_state=2020, shuffle=True)
    test_pred=np.empty((0,1),float)
    train_pred=np.empty((0,1),float)
    train_y = np.empty((0,1),float)
    fold_cnt = 0
    for train_indices,val_indices in folds.split(train_X_word,train_Y):
        fold_cnt+=1
        print(val_indices[[0,-1]])
        x_train_word, x_train_feature, x_val_word, x_val_feature = train_X_word[train_indices], train_X_feature[train_indices], train_X_word[val_indices], train_X_feature[val_indices]
        y_train,y_val=train_Y[train_indices],train_Y[val_indices]
        train_y = np.append(train_y, y_val)
        model_nn_fold = create_model_nn()
        callbacks_list = [
        keras.callbacks.EarlyStopping(
            monitor='val_accuracy',
            patience=5
        ),
        keras.callbacks.ModelCheckpoint(
            filepath='model/model_nn_p'+ str(fold_cnt) +'.h5',
            monitor='val_accuracy',
            save_best_only=True
        ),
        keras.callbacks.ReduceLROnPlateau(
            monitor='val_loss',
            factor=0.5,
            patience=5
        )
        ]
        model_nn_fold.fit([x_train_word, x_train_feature],y_train, epochs=epochs, batch_size=batch_size,
                          callbacks=callbacks_list, validation_data=([x_val_word, x_val_feature], y_val))
        model_nn_fold = load_model('model/model_nn_p'+ str(fold_cnt) +'.h5')
        prediction = model_nn_fold.predict([x_val_word, x_val_feature], batch_size=batch_size, verbose=True)
        train_pred=np.append(train_pred, np.array(list(chain.from_iterable(prediction.tolist()))))
    test_pred=np.append(test_pred,model.predict([test_X_word, test_X_feature], batch_size=batch_size, verbose=True))
    return train_pred, train_y, test_pred
