#output feature importance 
def feature_importance(type):
    imp = gbm.feature_importance(importance_type = type)
    impdf = pd.DataFrame(imp, columns = ['feature_imp'])
    imp_sort = impdf.sort_values(['feature_imp'], ascending = 0)
    imptop30 = imp_sort.iloc[0:30]
    key_list = list(tvec.vocabulary_.keys()) 
    val_list = list(tvec.vocabulary_.values()) 
    imptop30['feature_name'] = 0
    a=0
    for i in imptop30.index.tolist(): 
        imptop30['feature_name'].iloc[a] = key_list[val_list.index(i)] 
        a +=1

    plt.figure(figsize=(40, 20))
    sns.set(font_scale = 3)
    sns.barplot(x="feature_name", y="feature_imp", data=imptop30)
    plt.title('LightGBM Features')
    plt.tight_layout()
    #plt.savefig('lgbm_importances-01.png')
    plt.show()
